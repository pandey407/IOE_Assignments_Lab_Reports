\newcommand{\grapha}[3]{
\begin{tikzpicture}[scale=2]
  \datavisualization [school book axes,
    visualize as smooth line/.list={one},
    y axis={label={#3}},
    x axis={label={$t$}},
    one={style={black}},  
  ]

    data [set=one,format=function] {
        var x : interval [-#1:3];
        func y = exp(-2* (\value x+#1));
    };
    info{
    \node at (-#1+.4,0.4) [anchor=south west] {#2};
    };
    \draw [] (-#1,0) -- (-#1,1);
    
\end{tikzpicture}
}
