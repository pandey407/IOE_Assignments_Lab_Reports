\newcommand*{\widthFigure}[3][0.8]{
    \begin{figure}[H]
        \centering
        \includegraphics[width=#1\linewidth]{../Figures/#2.pdf}
    \label{fig:#2}
    \caption{#3}
    \end{figure}
}
